# Internationalization (i18n) Plan

## Overview

Add multi-language support to the frontend using `vue-i18n`. The default language is English. Users switch languages from a new selector inside the TopBar dropdown. The selected language persists across sessions via `localStorage`.

---

## Library

**`vue-i18n` v9** (the Vue 3-compatible release). It integrates natively with Vue's plugin system, supports Composition API via `useI18n()`, and is the ecosystem standard.

```
npm install vue-i18n@9
```

---

## File / Module Structure

```
frontend/src/
├── i18n/
│   ├── index.ts            # createI18n instance + export

└── components/
```

All other existing view and component files will have their hardcoded strings replaced with `t('key')` calls.

## Phase 1 — Setup

### 1.1 Install `vue-i18n`

```bash
cd frontend && npm install vue-i18n@9
```

### 1.2 Create the i18n instance (`frontend/src/i18n/index.ts`)

```ts
import { createI18n } from 'vue-i18n'
import en from './locales/en.json'
import fr from './locales/fr.json'
import cs from './locales/cs.json'

const STORAGE_KEY = 'locale'
const DEFAULT_LOCALE = 'en'

const savedLocale = localStorage.getItem(STORAGE_KEY) ?? DEFAULT_LOCALE

export const i18n = createI18n({
  legacy: false,           // use Composition API mode
  locale: savedLocale,
  fallbackLocale: 'en',
  messages: { en, fr, cs },
})
```

### 1.3 Register plugin (`frontend/src/main.ts`)

```ts
import { i18n } from './i18n'
app.use(i18n)
```

## Phase 2 — Locale Store

**`frontend/src/stores/locale.ts`**

Wraps `vue-i18n`'s `locale` ref so the rest of the app can switch languages through a single Pinia action (which also persists to `localStorage`).

```ts
import { defineStore } from 'pinia'
import { i18n } from '@/i18n'

export const SUPPORTED_LOCALES = [
  { code: 'en', label: 'English' },
  { code: 'fr', label: 'Français' },
  { code: 'cs', label: 'Čeština' },
] as const

export type LocaleCode = (typeof SUPPORTED_LOCALES)[number]['code']

export const useLocaleStore = defineStore('locale', () => {
  const current = i18n.global.locale   // ref<string> in Composition API mode

  function setLocale(code: LocaleCode) {
    current.value = code
    localStorage.setItem('locale', code)
  }

  return { current, setLocale, SUPPORTED_LOCALES }
})
```

## Phase 6 — Pre-commit Locale Key Check

Add a pre-commit hook that compares every non-English locale JSON file against the English file and exits non-zero if any keys differ (missing or extra). This prevents a locale from falling silently out of sync when new keys are added or old ones renamed.

### 6.1 Write the check script (`scripts/check-locales.py`)

Placed alongside the existing `scripts/check-requirements.sh`.

```python
#!/usr/bin/env python3
"""Verify that every locale JSON file has exactly the same keys as en.json."""

import json
import sys
from pathlib import Path


LOCALES_DIR = Path(__file__).parent.parent / "frontend" / "src" / "i18n" / "locales"


def flat_keys(obj: dict, prefix: str = "") -> set[str]:
    keys = set()
    for k, v in obj.items():
        full = f"{prefix}.{k}" if prefix else k
        if isinstance(v, dict):
            keys |= flat_keys(v, full)
        else:
            keys.add(full)
    return keys


def main() -> int:
    en = json.loads((LOCALES_DIR / "en.json").read_text())
    en_keys = flat_keys(en)

    failed = False
    for path in sorted(LOCALES_DIR.glob("*.json")):
        if path.name == "en.json":
            continue
        locale = json.loads(path.read_text())
        locale_keys = flat_keys(locale)

        missing = sorted(en_keys - locale_keys)
        extra = sorted(locale_keys - en_keys)

        if missing or extra:
            print(f"\n[{path.name}]")
            if missing:
                print("  Missing keys:\n" + "\n".join(f"    - {k}" for k in missing))
            if extra:
                print("  Extra keys:\n" + "\n".join(f"    + {k}" for k in extra))
            failed = True

    if failed:
        print("\nLocale check failed. Sync all locale files with en.json before committing.")
        return 1

    print("Locale check passed.")
    return 0


if __name__ == "__main__":
    sys.exit(main())
```

### 6.2 Register the hook in `.pre-commit-config.yaml`

Add a new entry under the existing `repo: local` block, following the same pattern as `check-requirements`:

```yaml
- id: check-locales
  name: Locale keys in sync
  entry: scripts/check-locales.py
  language: script
  pass_filenames: false
  files: ^frontend/src/i18n/locales/
```

The hook only runs when a file under `frontend/src/i18n/locales/` is staged. If any locale file is missing or has extra keys relative to `en.json`, the commit is blocked and the differing keys are printed.

## Phase 8 — Migrate `check-requirements.sh` to Python

Replace `scripts/check-requirements.sh` with `scripts/check-requirements.py` for better portability (no bash required on Windows or minimal container images).

### 8.1 Write the replacement script (`scripts/check-requirements.py`)

Replicates the same logic as the shell script: verify the venv exists, run `pip freeze`, and compare the sorted output to `requirements.txt`.

```python
#!/usr/bin/env python3
"""Verify that backend/requirements.txt matches the installed venv packages."""

import subprocess
import sys
from pathlib import Path

VENV_PIP = Path("backend/.venv-backend/bin/pip")
REQUIREMENTS = Path("backend/requirements.txt")


def main() -> int:
    if not VENV_PIP.exists():
        print(f"ERROR: Virtual environment not found at {VENV_PIP}")
        print("Create it with: cd backend && python -m venv .venv-backend && .venv-backend/bin/pip install -r requirements.txt")
        return 1

    frozen = subprocess.run(
        [str(VENV_PIP), "freeze"],
        capture_output=True, text=True, check=True,
    ).stdout

    frozen_pkgs = sorted(line.lower() for line in frozen.splitlines() if line)
    req_pkgs = sorted(line.lower() for line in REQUIREMENTS.read_text().splitlines() if line)

    if frozen_pkgs != req_pkgs:
        print("\nERROR: backend/requirements.txt is out of sync with .venv-backend.")
        print("Run:  cd backend && .venv-backend/bin/pip freeze > requirements.txt")
        return 1

    return 0


if __name__ == "__main__":
    sys.exit(main())
```

### 8.2 Update `.pre-commit-config.yaml`

Replace the `check-requirements` hook entry:

```yaml
- id: check-requirements
  name: Backend requirements in sync
  entry: scripts/check-requirements.py
  language: script
  pass_filenames: false
  always_run: true
```

Delete `scripts/check-requirements.sh` once the Python script is verified.

## Decisions & Notes

- **`legacy: false`** — use Composition API mode throughout; avoids `$t` on the Vue instance and plays well with `<script setup>`.
- **`localStorage` persistence** — the saved locale is read at i18n instantiation time (before Vue mounts), so there is no flash of the wrong language on reload.
- **Fallback to English** — any key missing from a non-English locale falls back to English automatically via `fallbackLocale: 'en'`.
- **No backend changes needed** — all text is frontend-only; API responses (error messages from the server) remain in English.
- **Adding more locales later** — add a new `.json` file under `locales/`, import it in `i18n/index.ts`, add an entry to `SUPPORTED_LOCALES` in the locale store, and the pre-commit hook will automatically enforce that the new file stays in sync with `en.json`.
- **Locale files as JSON** — JSON is used throughout; `vue-i18n` handles JSON imports natively and the Python check script parses them with the stdlib `json` module — no extra dependencies.
